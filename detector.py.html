import os
import time
import threading
import datetime
from database import insert_alert

folder_path = "test_folder"

alerts = []
running = False


def add_alert(message):
    print(message)
    alerts.append(message)

    timestamp = str(datetime.datetime.now())
    insert_alert(message, timestamp)


def monitor_folder():
    global running

    previous_locked = set()

    while running:
        try:
            current_files = os.listdir(folder_path)

            current_locked = set(
                [f for f in current_files if f.endswith(".locked")]
            )

            new_locked = current_locked - previous_locked

            for file in new_locked:
                add_alert(f"{datetime.datetime.now()} 🚨 Detected: {file}")

            previous_locked = current_locked

        except Exception as e:
            print("Error:", e)

        time.sleep(1)


def start_monitoring():
    global running
    if not running:
        running = True
        thread = threading.Thread(target=monitor_folder)
        thread.daemon = True
        thread.start()


def stop_monitoring():
    global running
    running = False
