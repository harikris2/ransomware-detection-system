from flask import Flask, render_template, request, redirect, session
from detector import start_monitoring, stop_monitoring
from database import init_db, get_alerts, register_user, login_user

# ONLY ADD (watchdog integration)
from watchdog_detector import start_watchdog, stop_watchdog

app = Flask(__name__)
app.secret_key = "secret123"

init_db()


@app.route("/")
def home():
    if "user" not in session:
        return redirect("/login")

    db_alerts = get_alerts()
    alerts_list = [row[1] for row in db_alerts]

    return render_template("index.html", alerts=alerts_list, user=session["user"])


# LOGIN
@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        username = request.form["username"]
        password = request.form["password"]

        user = login_user(username, password)

        if user:
            session["user"] = username
            return redirect("/")
        else:
            return "Invalid login"

    return render_template("login.html")


# REGISTER
@app.route("/register", methods=["GET", "POST"])
def register():
    if request.method == "POST":
        username = request.form["username"]
        password = request.form["password"]

        success = register_user(username, password)

        if success:
            return redirect("/login")
        else:
            return "User already exists"

    return render_template("register.html")


@app.route("/logout")
def logout():
    session.pop("user", None)
    return redirect("/login")


# OLD DETECTOR (UNCHANGED)
@app.route("/start")
def start():
    start_monitoring()
    return redirect("/")


@app.route("/stop")
def stop():
    stop_monitoring()
    return redirect("/")


# ✅ WATCHDOG (ONLY ADDITION)
@app.route("/watchdog-start")
def watchdog_start():
    start_watchdog()
    return redirect("/")


@app.route("/watchdog-stop")
def watchdog_stop():
    stop_watchdog()
    return redirect("/")


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=10000)
