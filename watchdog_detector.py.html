from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
import time
import datetime
from database import insert_alert

folder_path = "test_folder"


class RansomwareHandler(FileSystemEventHandler):

    def on_created(self, event):
        if not event.is_directory:
            message = f"{datetime.datetime.now()} 🚨 CREATED: {event.src_path}"
            print(message)
            insert_alert(message, str(datetime.datetime.now()))

    def on_modified(self, event):
        if not event.is_directory:
            message = f"{datetime.datetime.now()} ⚠ MODIFIED: {event.src_path}"
            print(message)
            insert_alert(message, str(datetime.datetime.now()))

    def on_deleted(self, event):
        if not event.is_directory:
            message = f"{datetime.datetime.now()} ❌ DELETED: {event.src_path}"
            print(message)
            insert_alert(message, str(datetime.datetime.now()))


observer = Observer()
observer.schedule(RansomwareHandler(), folder_path, recursive=False)


def start_watchdog():
    observer.start()
    print("Watchdog started...")


def stop_watchdog():
    observer.stop()
    observer.join()
    print("Watchdog stopped")
